03 · SQL Analysis

Business Questions
Answered in SQL

Key queries powering the analysis — segmentation logic, revenue breakdowns, and subscriber comparisons.

customer_behavior_project.sql 
-- Customer Segmentation: New / Returning / Loyal
SELECT
  customer_id,
  COUNT(*) AS purchase_count,
  SUM(purchase_amount_usd) AS total_spend,
  CASE
    WHEN COUNT(*) = 1               THEN 'New'
    WHEN COUNT(*) BETWEEN 2 AND 4  THEN 'Returning'
    ELSE                             'Loyal'
  END AS segment
FROM customer_behavior
GROUP BY customer_id
ORDER BY total_spend DESC;

-- Subscriber vs Non-Subscriber Average Spend
SELECT
  subscription_status,
  ROUND(AVG(purchase_amount_usd)::numeric, 2) AS avg_spend,
  COUNT(*)                                     AS customer_count
FROM customer_behavior
GROUP BY subscription_status;

-- High-Value Discount Users (top 10%)
SELECT customer_id, age, gender, purchase_amount_usd
FROM   customer_behavior
WHERE  discount_applied = 'Yes'
  AND  purchase_amount_usd > (
    SELECT PERCENTILE_CONT(0.9)
           WITHIN GROUP (ORDER BY purchase_amount_usd)
    FROM customer_behavior
  )
ORDER BY purchase_amount_usd DESC;
